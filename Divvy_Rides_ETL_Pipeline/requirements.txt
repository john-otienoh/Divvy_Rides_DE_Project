#!/usr/bin/env python3
"""
Scrape the Divvy Tripdata S3 index page and store file metadata in a database.
Supports both SQLite and DuckDB.
"""

import re
import sys
import logging
from datetime import datetime
from pathlib import Path
from typing import List, Dict, Optional

import requests
from bs4 import BeautifulSoup
import duckdb  # pip install duckdb
import sqlite3  # standard library

# ----------------------------------------------------------------------
# Configuration
# ----------------------------------------------------------------------
INDEX_URL = "https://divvy-tripdata.s3.amazonaws.com/index.html"
DB_TYPE = "duckdb"          # choose 'duckdb' or 'sqlite'
DB_PATH = "divvy_files.db"  # or ':memory:' for in-memory
LOG_FILE = "scrape.log"

# Regex to parse size strings like "3.32 MB", "14.73 MB", "6 KB"
SIZE_PATTERN = re.compile(r'([\d.]+)\s*(KB|MB|GB|TB|B)', re.IGNORECASE)

# ----------------------------------------------------------------------
# Logging
# ----------------------------------------------------------------------
def setup_logging():
    logging.basicConfig(
        level=logging.INFO,
        format="%(asctime)s - %(levelname)s - %(message)s",
        handlers=[
            logging.FileHandler(LOG_FILE, encoding='utf-8'),
            logging.StreamHandler(sys.stdout)
        ]
    )

# ----------------------------------------------------------------------
# Parse size string to bytes
# ----------------------------------------------------------------------
def size_to_bytes(size_str: str) -> Optional[int]:
    """Convert '3.32 MB' to bytes (int). Returns None if parsing fails."""
    match = SIZE_PATTERN.search(size_str)
    if not match:
        return None
    value = float(match.group(1))
    unit = match.group(2).upper()
    multipliers = {
        'B': 1,
        'KB': 1024,
        'MB': 1024**2,
        'GB': 1024**3,
        'TB': 1024**4,
    }
    return int(value * multipliers[unit])

# ----------------------------------------------------------------------
# Fetch and parse the index page
# ----------------------------------------------------------------------
def fetch_file_list(url: str) -> List[Dict]:
    """Return list of dicts with keys: name, last_modified, size_raw, size_bytes, type."""
    logging.info(f"Fetching {url}")
    resp = requests.get(url, timeout=30)
    resp.raise_for_status()

    soup = BeautifulSoup(resp.text, 'lxml')
    table = soup.find('table')
    if not table:
        raise ValueError("No <table> found in the HTML")

    # The table has <thead> and <tbody>. We'll target <tbody> rows.
    tbody = table.find('tbody')
    if not tbody:
        raise ValueError("No <tbody> found in table")

    rows = tbody.find_all('tr')
    if not rows:
        raise ValueError("No rows in table body")

    files = []
    for row in rows:
        cells = row.find_all('td')
        if len(cells) < 4:
            continue

        # Extract data
        name_cell = cells[0]
        # The <a> tag contains the filename
        a_tag = name_cell.find('a')
        if not a_tag:
            continue
        name = a_tag.text.strip()

        # Last modified is in cells[1]
        date_str = cells[1].text.strip()
        # Size is in cells[2]
        size_str = cells[2].text.strip()
        # Type is in cells[3]
        file_type = cells[3].text.strip()

        size_bytes = size_to_bytes(size_str)

        files.append({
            'name': name,
            'last_modified': date_str,
            'size_raw': size_str,
            'size_bytes': size_bytes,
            'type': file_type,
        })

    logging.info(f"Extracted {len(files)} file entries")
    return files

# ----------------------------------------------------------------------
# Database helpers
# ----------------------------------------------------------------------
def init_duckdb(db_path: str):
    """Create the files table in DuckDB."""
    conn = duckdb.connect(db_path)
    conn.execute("""
        CREATE TABLE IF NOT EXISTS files (
            name VARCHAR,
            last_modified VARCHAR,
            size_raw VARCHAR,
            size_bytes BIGINT,
            type VARCHAR,
            scraped_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    """)
    return conn

def insert_duckdb(conn, files: List[Dict]):
    """Insert files into DuckDB, replacing old data if needed."""
    conn.execute("DELETE FROM files")  # optional: replace all
    for f in files:
        conn.execute("""
            INSERT INTO files (name, last_modified, size_raw, size_bytes, type)
            VALUES (?, ?, ?, ?, ?)
        """, (f['name'], f['last_modified'], f['size_raw'], f['size_bytes'], f['type']))
    logging.info(f"Inserted {len(files)} rows into DuckDB")

def init_sqlite(db_path: str):
    """Create the files table in SQLite."""
    conn = sqlite3.connect(db_path)
    conn.execute("""
        CREATE TABLE IF NOT EXISTS files (
            name TEXT,
            last_modified TEXT,
            size_raw TEXT,
            size_bytes INTEGER,
            type TEXT,
            scraped_at TEXT DEFAULT (datetime('now'))
        )
    """)
    return conn

def insert_sqlite(conn, files: List[Dict]):
    """Insert files into SQLite."""
    conn.execute("DELETE FROM files")
    for f in files:
        conn.execute("""
            INSERT INTO files (name, last_modified, size_raw, size_bytes, type)
            VALUES (?, ?, ?, ?, ?)
        """, (f['name'], f['last_modified'], f['size_raw'], f['size_bytes'], f['type']))
    conn.commit()
    logging.info(f"Inserted {len(files)} rows into SQLite")

# ----------------------------------------------------------------------
# Main
# ----------------------------------------------------------------------
def main():
    setup_logging()
    logging.info("Starting Divvy index scraper")

    # 1. Fetch and parse
    try:
        files = fetch_file_list(INDEX_URL)
    except Exception as e:
        logging.error(f"Failed to fetch/parse index: {e}")
        sys.exit(1)

    # 2. Store in chosen database
    if DB_TYPE.lower() == 'duckdb':
        conn = init_duckdb(DB_PATH)
        insert_duckdb(conn, files)
        # Quick demo query
        result = conn.execute("SELECT COUNT(*) FROM files").fetchone()
        logging.info(f"Total files in DB: {result[0]}")
        conn.close()
    elif DB_TYPE.lower() == 'sqlite':
        conn = init_sqlite(DB_PATH)
        insert_sqlite(conn, files)
        cursor = conn.execute("SELECT COUNT(*) FROM files")
        logging.info(f"Total files in DB: {cursor.fetchone()[0]}")
        conn.close()
    else:
        logging.error(f"Unknown DB_TYPE: {DB_TYPE}")
        sys.exit(1)

    logging.info("Done. Data stored in %s", DB_PATH)

if __name__ == "__main__":
    main()
